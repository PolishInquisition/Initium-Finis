non_communist_or_capitulated_countries_asia = {
	input = game:all_countries
	name = COLLECTION_COUNTRIES_ASIA_NON_COMMUNIST
	operators = {
		limit = {
			capital_scope = { is_on_continent = asia }
			OR = {
				NOT = { has_government = communism }
				AND = {
					has_capitulated = yes 
					has_government = communism
				}
			}
		}
	}
